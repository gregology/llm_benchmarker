"""Legacy utilities for handling batch jobs."""
import json
import os
from typing import Dict, List

DEFAULT_TIMEOUT = 30
CONFIG = {'timeout': DEFAULT_TIMEOUT, 'retries': 3}

def load_config(path=None):
    path = path or os.getenv('JOB_CONFIG', 'config.json')
    if not os.path.exists(path):
        return CONFIG
    with open(path, 'r') as handle:
        data = json.load(handle)
    CONFIG.update(data)
    return CONFIG

def validate_items(items: List[Dict]):
    for item in items:
        if 'id' not in item:
            raise ValueError('missing id')
        if 'payload' not in item:
            raise ValueError('missing payload')

def process_item(item: Dict) -> Dict:
    if item.get('skip'):
        return {'id': item['id'], 'status': 'skipped'}
    payload = item['payload']
    return {'id': item['id'], 'status': 'ok', 'size': len(payload)}

def process_batch(items: List[Dict]) -> List[Dict]:
    load_config()
    validate_items(items)
    results = []
    for item in items:
        results.append(process_item(item))
    return results
